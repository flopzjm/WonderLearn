### Narración de Cuentos

#### Cuento 1: "El gato simpático"

```

Página 2: "El gato se llamaba Milo."
Página 3: "Milo vivía en una casa grande."
Página 4: "A Milo le gustaba jugar con una bola."
Página 5: "Un día, Milo encontró un ratón."
Página 6: "¡Pero eran amigos!"
```

**Configuración ElevenLabs:**
- Voz: "Spanish (Latin American) - Warm Female"
- Velocidad: 0.85x (lento para niños)
- Estabilidad: 0.75
- Claridad: 0.80

**Script para generar con Python (Bark):**
```python
from bark import SAMPLE_RATE, generate_audio, preload_models
import scipy.io.wavfile as wavfile

preload_models()

pages = [
    "Había una vez un gato muy simpático.",
    "El gato se llamaba Milo.",
    "Milo vivía en una casa grande.",
    "A Milo le gustaba jugar con una bola.",
    "Un día, Milo encontró un ratón.",
    "¡Pero eran amigos!",
]

for i, text in enumerate(pages, 1):
    audio_array = generate_audio(
        text, 
        history_prompt="v2/es_speaker_0"  # Voz femenina español
    )
    wavfile.write(f"story_01/page_{i}.wav", SAMPLE_RATE, audio_array)
    print(f"Página {i} generada")
```

## 📦 Script de Generación Automatizada

```python
#!/usr/bin/env python3
"""
WonderLearn Asset Generator
Genera assets placeholder usando IA
"""

import os
from pathlib import Path

# Estructura de carpetas
ASSETS_DIR = Path("wonderlearn_app/assets")
DIRECTORIES = [
    "images/letters",
    "images/numbers",
    "images/shapes",
    "images/emotions",
    "images/stories",
    "images/sight_words",
    "images/ui",
    "images/ui/icons",
    "audio/sfx",
    "audio/music",
    "audio/narration",
    "rive",
    "fonts",
]

def create_directories():
    """Crear estructura de carpetas"""
    for dir_path in DIRECTORIES:
        full_path = ASSETS_DIR / dir_path
        full_path.mkdir(parents=True, exist_ok=True)
        print(f"✅ Created: {full_path}")

def generate_prompts_file():
    """Generar archivo de prompts para Midjourney/DALL-E"""
    prompts = {
        "letters": [],
        "stories": [],
        "emotions": [],
    }
    
    # Prompts para letras
    letters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    words = {
        "A": ["Árbol", "Avión", "Ardilla"],
        "B": ["Bola", "Casa", "Barco"],
        # ... completar
    }
    
    for letter in letters:
        prompts["letters"].append({
            "letter": letter,
            "prompt": f"Large letter {letter} with {words[letter][0].lower()}, kawaii style, educational, pastel, white background"
        })
    
    # Guardar prompts
    import json
    with open("prompts.json", "w") as f:
        json.dump(prompts, f, indent=2, ensure_ascii=False)
    
    print("✅ Prompts generated: prompts.json")

if __name__ == "__main__":
    print("🎨 WonderLearn Asset Generator")
    print("=" * 40)
    create_directories()
    generate_prompts_file()
    print("=" * 40)
    print("✅ Done! Next steps:")
    print("1. Use prompts.json with Midjourney/DALL-E")
    print("2. Generate audio with ElevenLabs/Bark")
    print("3. Create animations with Lottiefiles AI")
```
